VOLDEX API
Your AI, your way. Simple, fast, powerful.
v1.0 Stable
Base URL
https://voldex.vercel.app/api
All requests must use HTTPS. The API endpoint is /api on your Vercel domain. No /v1 prefix.
Authentication

Every request requires a valid Voldex API key in the Authorization header.

Endpoints
POST /api
The main endpoint for AI chat. Returns a streaming response using Server-Sent Events (SSE). Accepts Gemini native format. 
// Request headers
Content-Type: application/json
Authorization: Bearer API-VOLDEX-xxxxxxxx

// Request body (Gemini native format)
{
  "systemInstruction": {
    "parts": [{ "text": "You are Voldex, a helpful AI assistant." }]
  },
  "contents": [
    {
      "role": "user",
      "parts": [{ "text": "Hello, who are you?" }]
    }
  ],
  "generationConfig": {
    "temperature": 0.7
  }
}
Field Type Required Description
systemInstruction object No System prompt with parts: [{text: "..."}]
contents array Yes Conversation history in Gemini format
contents[].role string Yes "user" or "model" (not "assistant")
contents[].parts array Yes Array of {text: "..."} objects
generationConfig object No {temperature: 0.0-2.0} (default: 0.7)

Code Examples
JavaScript (Fetch) — Correct Format
const API_KEY = 'API-VOLDEX-xxxxxxxx';
const API_URL = 'https://voldex.vercel.app/api';

async function chat(message) {
  const response = await fetch(API_URL, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${API_KEY}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      systemInstruction: { parts: [{ text: 'You are Voldex.' }] },
      contents: [{ role: 'user', parts: [{ text: message }] }],
      generationConfig: { temperature: 0.7 }
    })
  });

  const reader = response.body.getReader();
  const decoder = new TextDecoder();
  let buffer = '';

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    buffer += decoder.decode(value, { stream: true });

    const lines = buffer.split('\n');
    buffer = lines.pop() || '';

    for (const line of lines) {
      if (line.startsWith('data: ')) {
        const data = line.slice(6);
        if (data === '[DONE]') continue;
        const parsed = JSON.parse(data);
        const text = parsed.candidates?.[0]?.content?.parts?.map(p => p.text).join('') || '';
        console.log(text);
      }
    }
  }
}

Python (Requests) — Correct Format
import requests
import json

API_KEY = 'API-VOLDEX-xxxxxxxx'
API_URL = 'https://voldex.vercel.app/api'

def chat(message):
    response = requests.post(API_URL, headers={
        'Authorization': f'Bearer {API_KEY}',
        'Content-Type': 'application/json'
    }, json={
        'systemInstruction': {'parts': [{'text': 'You are Voldex.'}]},
        'contents': [{'role': 'user', 'parts': [{'text': message}]}],
        'generationConfig': {'temperature': 0.7}
    }, stream=True)

    for line in response.iter_lines():
        if line.startswith(b'data: '):
            data = line[6:].decode()
            if data == '[DONE]': continue
            parsed = json.loads(data)
            text = ''.join(p.get('text', '') for p in parsed.get('candidates', [{}])[0].get('content', {}).get('parts', []))
            print(text, end='', flush=True)

Python (urllib — no requests needed)
import urllib.request
import json
import ssl

API_KEY = 'API-VOLDEX-xxxxxxxx'
API_URL = 'https://voldex.vercel.app/api'

payload = {
    'systemInstruction': {'parts': [{'text': 'You are Voldex.'}]},
    'contents': [{'role': 'user', 'parts': [{'text': 'Hello'}]}]
}

data = json.dumps(payload).encode('utf-8')
req = urllib.request.Request(API_URL, data=data, method='POST')
req.add_header('Authorization', f'Bearer {API_KEY}')
req.add_header('Content-Type', 'application/json')

ctx = ssl.create_default_context()
with urllib.request.urlopen(req, context=ctx) as response:
    for line in response:
        line = line.decode('utf-8').strip()
        if line.startswith('data: '):
            data = line[6:]
            if data == '[DONE]': continue
            parsed = json.loads(data)
            text = ''.join(p.get('text', '') for p in parsed.get('candidates', [{}])[0].get('content', {}).get('parts', []))
            print(text, end='')
Important Notes
Format: This API uses Gemini native format, not OpenAI format. Use contents with role: "user" or role: "model" and parts: [{text: "..."}].
Streaming: Responses are Server-Sent Events (SSE). Each line starts with data:. Parse as JSON, extract candidates[0].content.parts[].text.
Images: Send base64-encoded images using inlineData: {mimeType: "image/png", data: "base64string"} inside parts.
Backend: Voldex rotates between Gemini backends automatically. If one is busy, it tries the next. You always get the fastest available response.
